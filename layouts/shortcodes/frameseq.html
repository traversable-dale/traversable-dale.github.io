{{- /* Canvas frame-sequence scroll scrubber (for rendered clips / image2lottie WebP sequences). Preloads a folder of frames and draws the current one to a based on scroll position — sharp, flicker-free, scroll-synced. Shows a "loading… NN%" indicator while frames download, and (in sticky mode) holds the page at the pin point until they're in. Usage: {{< frameseq dir="/lottie/cat-frames" count="423" >}} {{< frameseq dir="/lottie/cat-frames" dir-small="/lottie/cat-frames-sm" count="423" >}} {{< frameseq dir="/lottie/cat-frames" count="423" sticky="false" height="480px" >}} Params: dir (required) folder under /static holding the frames (leading /) count (required) number of frames dir-small optional lower-resolution frame folder used on narrow screens — this is what keeps mobile smooth (fewer pixels to decode per draw) small-max viewport width (px) at or below which dir-small is used (default 768) pad zero-padding width of the frame index (default 3 -> frame_000) ext frame file extension (default webp) sticky "true" (default) pins the canvas full-height and scrubs as you scroll past; "false" scrubs inline as the box crosses the viewport track scroll distance for a full play-through in sticky mode (default 300vh) height canvas height in inline mode (default 70vh) label text shown while loading (default "loading") class extra CSS classes */ -}} {{- $dir := .Get "dir" -}} {{- $count := .Get "count" -}} {{- if or (not $dir) (not $count) -}} {{- errorf "shortcode frameseq: 'dir' and 'count' are required at %s" .Position -}} {{- else -}} {{- $dirSmall := .Get "dir-small" | default "" -}} {{- $smallMax := .Get "small-max" | default "768" -}} {{- $pad := .Get "pad" | default "3" -}} {{- $ext := .Get "ext" | default "webp" -}} {{- $sticky := ne (.Get "sticky" | default "true") "false" -}} {{- $track := .Get "track" | default "300vh" -}} {{- $height := .Get "height" | default "70vh" -}} {{- $label := .Get "label" | default "loading" -}} {{- $class := .Get "class" | default "" -}} {{- $attrs := printf `data-frame-dir="%s" data-frame-count="%s" data-frame-pad="%s" data-frame-ext="%s" data-frame-small-max="%s" data-frame-lock="%t"` ($dir | relURL) $count $pad $ext $smallMax $sticky -}} {{- with $dirSmall -}} {{- $attrs = printf `%s data-frame-dir-small="%s"` $attrs ($dirSmall | relURL) -}} {{- end -}} {{- $loader := printf `
%s... 0%%
` $label -}} {{- if $sticky -}}
{{ $loader | safeHTML }}
{{- else -}}
{{ $loader | safeHTML }}
{{- end -}} {{- end -}}
